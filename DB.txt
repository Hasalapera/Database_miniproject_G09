Student Table

CREATE TABLE Students (
    Student_ID VARCHAR(6) PRIMARY KEY,
    Stu_FirstName VARCHAR(50),
    Stu_LastName VARCHAR(50),
    Stu_DateOfBirth DATE,
    Stu_Address VARCHAR(100),
    Stu_Email VARCHAR(100),
    Stu_PhoneNumber VARCHAR(10),
    Stu_EnrollmentDate DATE,
    Status VARCHAR(10) -- Proper, Repeat, Suspended
    Stu_DepID CHAR(3),
);
